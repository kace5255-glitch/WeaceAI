## 一、現有系統的瓶頸診斷

先看你目前的上下文構建邏輯（`App.tsx` 第 202-221 行的 `buildContextFromChapters`）：

```typescript
// 每個參考章節 = briefing全文 + 最後1000字內容 + 大綱(fallback)

    return chapters.map(chap => {

        if (chap.content) parts.push(`[${chap.title} 結尾片段]:\n${chap.content.slice(-1000)}`);

```

再看 `buildUserPrompt`（`server/index.js` 第 314 行）：當前章節內容取最後 6000 字。

問題很明確：

1. 用戶手動選參考章節，選 5 章就是 5 份 briefing + 5000 字尾段，大約 8000-15000 字，加上 system prompt 的寫作規範（約 3000-5000 字）、角色/詞彙/世界觀，一次請求輕鬆超過 30K tokens
2. 沒有任何機制讓 AI 知道第 50 章埋的伏筆 -- 除非用戶手動選第 50 章作為參考
3. briefing 是扁平的章節摘要，沒有結構化的語義索引，無法按「伏筆」「角色狀態」「時間線」等維度查詢
4. 備忘錄（memos）是純文字，沒有和章節/角色關聯，也不會自動注入 prompt


## 二、多層記憶架構設計

核心思路：模仿人類記憶的「工作記憶 / 短期記憶 / 長期記憶」三層結構，每層有不同的壓縮比和檢索策略。

### 2.1 數據結構設計

```sql
-- ═══ 新增表：結構化記憶系統 ═══

-- 1. 章節事件索引（從 briefing 中提取的結構化數據）
CREATE TABLE IF NOT EXISTS chapter_events (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    novel_id UUID REFERENCES novels(id) ON DELETE CASCADE,
    chapter_id UUID REFERENCES chapters(id) ON DELETE CASCADE,
    chapter_index INTEGER NOT NULL,          -- 第幾章（全書序號）
    event_type TEXT NOT NULL,                -- 'plot' | 'foreshadow' | 'character_change' | 'power_up' | 'relationship' | 'worldbuilding' | 'conflict' | 'resolution'
    summary TEXT NOT NULL,                   -- 事件摘要（50-100字）
    involved_characters TEXT[],              -- 涉及角色名
    tags TEXT[],                             -- 自由標籤，如 ['伏筆:未回收', '主線', '感情線']
    importance INTEGER DEFAULT 5,            -- 1-10 重要度
    resolved BOOLEAN DEFAULT FALSE,          -- 伏筆是否已回收
    resolved_at_chapter INTEGER,             -- 在第幾章回收
    created_at TIMESTAMP DEFAULT NOW()
);

-- 2. 角色狀態時間線（角色在不同章節的狀態快照）
CREATE TABLE IF NOT EXISTS character_timeline (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    novel_id UUID REFERENCES novels(id) ON DELETE CASCADE,
    character_id UUID REFERENCES characters(id) ON DELETE CASCADE,
    chapter_id UUID REFERENCES chapters(id) ON DELETE CASCADE,
    chapter_index INTEGER NOT NULL,
    status_snapshot JSONB NOT NULL,          -- { location, power_level, emotional_state, relationships: {name: relation}, injuries, possessions }
    key_action TEXT,                         -- 本章關鍵行為
    created_at TIMESTAMP DEFAULT NOW()
);

-- 3. 分卷摘要（Volume-level 壓縮記憶）
CREATE TABLE IF NOT EXISTS volume_summaries (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    novel_id UUID REFERENCES novels(id) ON DELETE CASCADE,
    volume_id UUID REFERENCES volumes(id) ON DELETE CASCADE,
    summary TEXT NOT NULL,                   -- 整卷劇情摘要（500-1000字）
    key_events TEXT[],                       -- 關鍵事件列表
    unresolved_threads TEXT[],               -- 未解決的伏筆/懸念
    character_arcs JSONB,                    -- { "角色名": "本卷成長弧線摘要" }
    created_at TIMESTAMP DEFAULT NOW()
);

-- 4. 全書記憶索引（最高層壓縮）
CREATE TABLE IF NOT EXISTS novel_memory_index (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    novel_id UUID REFERENCES novels(id) ON DELETE CASCADE,
    memory_type TEXT NOT NULL,               -- 'timeline' | 'power_system' | 'faction_map' | 'relationship_web' | 'unresolved_foreshadow' | 'recurring_theme'
    content TEXT NOT NULL,                   -- 結構化文本
    last_updated_chapter INTEGER,            -- 最後更新到第幾章
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_chapter_events_novel ON chapter_events(novel_id);
CREATE INDEX idx_chapter_events_type ON chapter_events(novel_id, event_type);
CREATE INDEX idx_chapter_events_tags ON chapter_events USING GIN(tags);
CREATE INDEX idx_chapter_events_characters ON chapter_events USING GIN(involved_characters);
CREATE INDEX idx_character_timeline_char ON character_timeline(character_id, chapter_index);
CREATE INDEX idx_volume_summaries_novel ON volume_summaries(novel_id);
CREATE INDEX idx_novel_memory_novel ON novel_memory_index(novel_id);
```

### 2.2 三層記憶的運作方式

```
┌─────────────────────────────────────────────────────────┐
│  Layer 3: 全書記憶 (novel_memory_index)                  │
│  壓縮比 ~500:1 | 整本書 → 2000-3000字                    │
│  內容：時間線、勢力圖、未回收伏筆清單、角色關係網          │
│  更新頻率：每 10-20 章或每卷結束時重新生成                 │
├─────────────────────────────────────────────────────────┤
│  Layer 2: 分卷摘要 (volume_summaries)                    │
│  壓縮比 ~50:1 | 每卷 → 500-1000字                        │
│  內容：卷級劇情弧線、角色成長、未解懸念                    │
│  更新頻率：每卷完結時生成                                  │
├─────────────────────────────────────────────────────────┤
│  Layer 1: 章節事件 (chapter_events)                      │
│  壓縮比 ~10:1 | 每章 → 3-8 條結構化事件                   │
│  內容：劇情事件、伏筆、角色變動、戰力變化                  │
│  更新頻率：每章 briefing 生成後自動提取                    │
├─────────────────────────────────────────────────────────┤
│  Layer 0: 原始內容 (chapters.content + briefing)         │
│  壓縮比 1:1 | 完整章節文本                                │
│  僅用於：當前章 + 前1-2章的尾段                           │
└─────────────────────────────────────────────────────────┘
```

### 3.2 智能上下文組裝演算法

這是核心。不再讓用戶手動選參考章節，而是系統自動組裝最相關的上下文：

```javascript
// server/contextBuilder.js — 新增的上下文組裝引擎

const TOKEN_BUDGET = {
    systemPrompt: 5000,
    worldview: 3000,
    characters: 2000,
    vocabulary: 1000,
    globalMemory: 3000,      // Layer 3
    volumeSummaries: 3000,   // Layer 2
    relevantEvents: 3000,    // Layer 1 (智能檢索)
    unresolvedForeshadow: 2000,
    characterSnapshots: 2000,
    recentBriefings: 4000,   // 前1-3章
    recentContent: 4000,     // 前1章尾段
    currentContent: 10000,   // 當前章
    userInstructions: 3000,
    outputReserve: 20000,
    flexible: 35000          // 彈性空間
};

async function buildSmartContext(novelId, currentChapterIndex, supabase) {
    const context = {};

    // 1. Layer 3: 全書記憶（永遠注入）
    const { data: globalMemories } = await supabase
        .from('novel_memory_index')
        .select('memory_type, content')
        .eq('novel_id', novelId);

    context.globalMemory = globalMemories
        ?.map(m => `【${m.memory_type}】\n${m.content}`)
        .join('\n\n') || '';

    // 2. Layer 2: 當前卷 + 前一卷的摘要
    const { data: volSummaries } = await supabase
        .from('volume_summaries')
        .select('*, volumes!inner(order_index)')
        .eq('novel_id', novelId)
        .order('volumes.order_index', { ascending: false })
        .limit(3);

    context.volumeSummaries = volSummaries
        ?.map(vs => `【${vs.volumes?.title || '卷'}摘要】\n${vs.summary}`)
        .join('\n\n') || '';

    // 3. Layer 1: 智能事件檢索
    // 3a. 未回收的伏筆（永遠注入）
    const { data: foreshadows } = await supabase
        .from('chapter_events')
        .select('chapter_index, summary, tags')
        .eq('novel_id', novelId)
        .eq('event_type', 'foreshadow')
        .eq('resolved', false)
        .order('importance', { ascending: false })
        .limit(15);

    context.unresolvedForeshadow = foreshadows
        ?.map(f => `- [第${f.chapter_index}章] ${f.summary}`)
        .join('\n') || '';

    // 3b. 涉及當前章登場角色的近期事件
    // (由前端傳入 selectedCharacterNames)
    // 3c. 最近 5 章的高重要度事件
    const { data: recentEvents } = await supabase
        .from('chapter_events')
        .select('*')
        .eq('novel_id', novelId)
        .gte('chapter_index', Math.max(1, currentChapterIndex - 20))
        .gte('importance', 7)
        .order('chapter_index', { ascending: false })
        .limit(20);

    context.relevantEvents = recentEvents
        ?.map(e => `[第${e.chapter_index}章|${e.event_type}] ${e.summary}`)
        .join('\n') || '';

    // 4. 角色最新狀態快照
    // (根據選中角色，取每個角色最新的 timeline entry)

    return context;
}
```

### 4.2 一致性檢查端點

在生成前或生成後，可以跑一次一致性檢查：

```javascript
app.post('/api/memory/consistency-check', async (req, res) => {
    const { novelId, chapterContent, chapterIndex } = req.body;

    // 拉取相關記憶
    const { data: charTimelines } = await supabase
        .from('character_timeline')
        .select('*, characters!inner(name)')
        .eq('novel_id', novelId)
        .order('chapter_index', { ascending: false });

        .from('chapter_events')

        .eq('event_type', 'foreshadow')
```

## 七、業界參考與借鑑

### NovelAI
- 使用 Lorebook（知識庫）機制：每個條目有觸發關鍵詞，當用戶輸入或生成內容包含關鍵詞時自動注入

### Sudowrite
- Story Bible：結構化的角色卡、地點卡、物品卡，每張卡有「出場章節」索引
- Beat Sheet：把整本書拆成 beats（節拍），每個 beat 是一個劇情單元

### 3.2 統一的 Token 預算分配表

```javascript
// server/tokenBudget.js

const MODEL_BUDGETS = {
    'Google Flash':  { total: 900000, output: 8000 },
    'Gemini 2.5':    { total: 900000, output: 16000 },
    'Gemini Pro':    { total: 900000, output: 16000 },
    'DeepSeek R1':   { total: 60000,  output: 8000 },
    'DeepSeek V3.2': { total: 60000,  output: 8000 },
    'Kimi':          { total: 120000, output: 8000 },
    'Claude':        { total: 190000, output: 8000 },
};

// 中文字 → token 的粗略換算：1 中文字 ≈ 1.5-2 tokens
const CHARS_TO_TOKENS = 1.8;

function allocateBudget(model, systemPromptChars) {
    const budget = MODEL_BUDGETS[model] || MODEL_BUDGETS['Google Flash'];
    const available = budget.total - budget.output;
    const systemTokens = Math.ceil(systemPromptChars * CHARS_TO_TOKENS);
    const contextBudget = available - systemTokens;

    // 固定分配（不隨模型變化）
    const fixed = {
        globalMemory:    3000,   // Layer 3 全書記憶
        volumeSummaries: 2500,   // Layer 2 當前卷+前卷摘要
        unresolvedForeshadow: 2000, // 未回收伏筆
        characterSnapshots: 1500,   // 角色最新狀態
        relevantEvents:  2500,   // Layer 1 智能檢索的事件
        recentBriefings: 3000,   // 前1-3章 briefing
        currentChapterTail: 10000, // 當前章已寫內容尾段
        userInstructions: 3000,  // 大綱+要求+關係
    };

    const fixedTotal = Object.values(fixed).reduce((a, b) => a + b, 0);
    // 剩餘空間 = 彈性空間，用於更多參考章節或更長的當前章內容
    const flexible = Math.max(0, contextBudget - fixedTotal);

    return { ...fixed, flexible, contextBudget, systemTokens };
}
```

```javascript
/**
 * 重建全書記憶索引
 * 把所有卷摘要 + 未回收伏筆 + 角色最新狀態壓縮成全書級索引
 */
async function rebuildNovelMemoryIndex(genAI, {
    novelTitle, genre, worldview,
    volumeSummaries, unresolvedForeshadows, characterLatestStates
}) {
    const model = genAI.getGenerativeModel({
        model: 'gemini-2.5-flash',
        generationConfig: { temperature: 0.1 }
    });

    const volText = volumeSummaries
        .map(vs => `【${vs.volume_title}】\n${vs.summary}`)
        .join('\n\n');

    const foreshadowText = unresolvedForeshadows
        .map(f => `[第${f.chapter_index}章|重要度${f.importance}] ${f.summary}`)
        .join('\n');

    const prompt = `你是小說《${novelTitle}》（${genre}）的總編輯。
請根據以下資料，生成全書記憶索引。這個索引會在每次 AI 生成新章節時注入，確保全書一致性。

【世界觀設定】
${worldview || '未設定'}

【各卷摘要】
${volText}

【未回收伏筆】
${foreshadowText}

【角色最新狀態】
${characterLatestStates.map(c => `${c.name}：位於${c.location}，${c.power_level}，${c.emotional_state}`).join('\n')}

請分別生成以下 7 個索引（每個控制在 500-1000 字以內）：

1. **timeline**：全書時間線（按時間順序列出關鍵事件節點）
2. **power_system**：當前的戰力/等級體系格局（誰強誰弱，最新排名）
3. **faction_map**：勢力版圖（各勢力的關係、領地、核心人物）
4. **relationship_web**：核心角色關係網（誰和誰是什麼關係，最新狀態）
5. **unresolved_foreshadow**：未回收伏筆總表（按重要度排序，標注埋設章節）
6. **recurring_theme**：反覆出現的主題和意象（用於保持風格一致）
7. **world_state**：世界當前狀態（最新的政治/社會/環境狀況）

每個索引用 --- 分隔，第一行標注類型名。`;

    const result = await model.generateContent({
        contents: [{ role: 'user', parts: [{ text: prompt }] }]
    });

    // 解析並拆分成 7 個索引
    const text = result.response.text();
    const sections = text.split(/---+/).filter(s => s.trim());
    const memoryTypes = ['timeline', 'power_system', 'faction_map',
        'relationship_web', 'unresolved_foreshadow', 'recurring_theme', 'world_state'];

    return memoryTypes.map((type, i) => ({
        memory_type: type,
        content: sections[i]?.trim() || ''
    })).filter(m => m.content);
}
```

// server/consistencyChecker.js

/**
 * 生成前的一致性預檢
 * 在 buildUserPrompt 之前調用，把潛在衝突點注入 prompt
 */
async function preGenerationCheck(supabase, genAI, {
    novelId, currentChapterIndex, outline, selectedCharacterNames
}) {
    // 1. 取選中角色的最新狀態
    const characterStates = {};
    for (const name of selectedCharacterNames) {
        const { data } = await supabase
            .from('character_timeline')
            .select('*')
            .eq('novel_id', novelId)
            .eq('character_name', name)
            .order('chapter_index', { ascending: false })
            .limit(1);
        if (data?.[0]) characterStates[name] = data[0];
    }

    // 2. 取未回收伏筆中涉及這些角色的
    const { data: relevantForeshadows } = await supabase
        .from('chapter_events')
        .select('*')
        .eq('novel_id', novelId)
        .eq('event_type', 'foreshadow')
        .is('linked_foreshadow_id', null)
        .overlaps('involved_characters', selectedCharacterNames);

    // 3. 組裝成「注意事項」文本，直接注入 userPrompt
    let warnings = [];

    for (const [name, state] of Object.entries(characterStates)) {
        if (state.injuries) {
            warnings.push(`⚠️ ${name} 在第${state.chapter_index}章受傷：${state.injuries}，請注意傷勢延續`);
        }
        if (state.location) {
            warnings.push(`📍 ${name} 最後出現在：${state.location}（第${state.chapter_index}章）`);
        }
    }

    if (relevantForeshadows?.length > 0) {
        warnings.push(`\n💡 可回收的伏筆：`);
        relevantForeshadows.forEach(f => {
            warnings.push(`  - [第${f.chapter_index}章] ${f.summary}`);
        });
    }

    return warnings.length > 0
        ? `\n【一致性提醒 — 系統自動檢測】\n${warnings.join('\n')}`
        : '';
}
